256 void refresh(bool compute_tangent = true)
257 {
258 namespace GT = GMatTensor::Cartesian3d::pointer;
259
260#pragma omp parallel
261 {
262 auto II = GMatTensor::Cartesian3d::II();
263 auto I4d = GMatTensor::Cartesian3d::I4d();
264 auto I4s = GMatTensor::Cartesian3d::I4s();
265 auto I4rt = GMatTensor::Cartesian3d::I4rt();
266 auto nI4rt = xt::eval(-1.0 * I4rt);
267
268 double K;
269 double G;
270
271 std::array<double, m_stride_tensor2> Be;
272 std::array<double, m_stride_tensor2> vec;
273 std::array<double, m_ndim> Be_val;
274 std::array<double, m_ndim> Eps_val;
275 std::array<double, m_ndim> Epsd_val;
276 std::array<double, m_ndim> Sig_val;
277
278 std::array<size_t, 4> tensor4 = {m_ndim, m_ndim, m_ndim, m_ndim};
279 array_type::tensor<double, 4> dTau_dlnBe = xt::empty<double>(tensor4);
280 array_type::tensor<double, 4> dlnBe_dBe = xt::empty<double>(tensor4);
281 array_type::tensor<double, 4> dBe_dLT = xt::empty<double>(tensor4);
282 array_type::tensor<double, 4> Kmat = xt::empty<double>(tensor4);
283 array_type::tensor<double, 4> Kgeo = xt::empty<double>(tensor4);
284
285 auto F = xt::adapt(m_F.data(), {m_ndim, m_ndim});
286 auto Sig = xt::adapt(m_Sig.data(), {m_ndim, m_ndim});
287 auto C = xt::adapt(m_C.data(), {m_ndim, m_ndim, m_ndim, m_ndim});
288
289#pragma omp for
290 for (size_t i = 0; i < m_size; ++i) {
291
292 K = m_K.flat(i);
293 G = m_G.flat(i);
294
295 F.reset_buffer(&m_F.flat(i * m_stride_tensor2), m_stride_tensor2);
296 Sig.reset_buffer(&m_Sig.flat(i * m_stride_tensor2), m_stride_tensor2);
297 C.reset_buffer(&m_C.flat(i * m_stride_tensor4), m_stride_tensor4);
298
299 // volume change ratio
300 double J = GT::Det(F.data());
301
302 // Finger tensor
303 GT::A2_dot_A2T(F.data(), &Be[0]);
304
305 // eigenvalue decomposition of the trial "Be"
306 GT::eigs(&Be[0], &vec[0], &Be_val[0]);
307
308 // logarithmic strain "Eps := 0.5 ln(Be)" (in diagonalised form)
309 for (size_t j = 0; j < 3; ++j) {
310 Eps_val[j] = 0.5 * std::log(Be_val[j]);
311 }
312
313 // decompose strain (in diagonalised form)
314 double epsm = (Eps_val[0] + Eps_val[1] + Eps_val[2]) / 3.0;
315 for (size_t j = 0; j < 3; ++j) {
316 Epsd_val[j] = Eps_val[j] - epsm;
317 }
318
319 // Cauchy stress (in diagonalised form)
320 for (size_t j = 0; j < 3; ++j) {
321 Sig_val[j] = (3.0 * K * epsm + 2.0 * G * Epsd_val[j]) / J;
322 }
323
324 // compute Cauchy stress, in original coordinate frame
325 GT::from_eigs(&vec[0], &Sig_val[0], Sig.data());
326
327 if (!compute_tangent) {
328 return;
329 }
330
331 // 'linearisation' of the constitutive response
332 // Use that "Tau := Ce : Eps = 0.5 * Ce : ln(Be)"
333 xt::noalias(dTau_dlnBe) = 0.5 * K * II + G * I4d;
334 dlnBe_dBe.fill(0.0);
335
336 for (size_t m = 0; m < 3; ++m) {
337 for (size_t n = 0; n < 3; ++n) {
338
339 double gc = 2.0 * (Eps_val[n] - Eps_val[m]) / (Be_val[n] - Be_val[m]);
340
341 if (Be_val[m] == Be_val[n]) {
342 gc = 1.0 / Be_val[m];
343 }
344
345 for (size_t i = 0; i < 3; ++i) {
346 for (size_t j = 0; j < 3; ++j) {
347 for (size_t k = 0; k < 3; ++k) {
348 for (size_t l = 0; l < 3; ++l) {
349 dlnBe_dBe(i, j, k, l) += gc * vec[i * 3 + m] *
350 vec[j * 3 + n] * vec[k * 3 + m] *
351 vec[l * 3 + n];
352 }
353 }
354 }
355 }
356 }
357 }
358
359 // linearization of "Be"
360 // Use that "dBe = 2 * (I4s . Be) : LT" (where "LT" refers to "L_\delta^T")
361 // Hence: "dBe_dLT = 2 * (I4s * Be)"
362 GT::A4_dot_B2(I4s.data(), &Be[0], dBe_dLT.data());
363 dBe_dLT *= 2.0;
364
365 // material tangent stiffness
366 // Kmat = dTau_dlnBe : dlnBe_dBe : dBe_dLT
367 GT::A4_ddot_B4_ddot_C4(
368 dTau_dlnBe.data(), dlnBe_dBe.data(), dBe_dLT.data(), Kmat.data());
369
370 // geometrically non-linear tangent
371 // Kgeo = -I4rt . Tau
372 GT::A4_dot_B2(nI4rt.data(), Sig.data(), Kgeo.data());
373
374 // combine tangents:
375 xt::noalias(C) = Kgeo + Kmat / J;
376 }
377 }
378 }

565 void refresh(bool compute_tangent = true)
566 {
567 namespace GT = GMatTensor::Cartesian3d::pointer;
568
569#pragma omp parallel
570 {
571 auto II = GMatTensor::Cartesian3d::II();
572 auto I4d = GMatTensor::Cartesian3d::I4d();
573 auto I4s = GMatTensor::Cartesian3d::I4s();
574 auto I4rt = GMatTensor::Cartesian3d::I4rt();
575 auto nI4rt = xt::eval(-1.0 * I4rt);
576
577 double K;
578 double G;
579 double tauy0;
580 double H;
581 double epsp_t;
582
583 std::array<double, m_stride_tensor2> Finv_t;
584 std::array<double, m_stride_tensor2> Fdelta;
585 std::array<double, m_stride_tensor2> Be_trial;
586 std::array<double, m_stride_tensor2> N2;
587 std::array<double, m_stride_tensor2> vec;
588 std::array<double, m_ndim> Be_trial_val;
589 std::array<double, m_ndim> Epse_val;
590 std::array<double, m_ndim> Epsed_val;
591 std::array<double, m_ndim> Taud_val;
592 std::array<double, m_ndim> Sig_val;
593 std::array<double, m_ndim> N_val;
594 std::array<double, m_ndim> lnBe_val;
595
596 std::array<size_t, 4> tensor4 = {m_ndim, m_ndim, m_ndim, m_ndim};
597 array_type::tensor<double, 4> NN = xt::empty<double>(tensor4);
598 array_type::tensor<double, 4> dTau_dlnBe = xt::empty<double>(tensor4);
599 array_type::tensor<double, 4> dlnBe_dBe = xt::empty<double>(tensor4);
600 array_type::tensor<double, 4> dBe_dLT = xt::empty<double>(tensor4);
601 array_type::tensor<double, 4> Kmat = xt::empty<double>(tensor4);
602 array_type::tensor<double, 4> Kgeo = xt::empty<double>(tensor4);
603
604 auto F = xt::adapt(m_F.data(), {m_ndim, m_ndim});
605 auto F_t = xt::adapt(m_F_t.data(), {m_ndim, m_ndim});
606 auto Be = xt::adapt(m_Be.data(), {m_ndim, m_ndim});
607 auto Be_t = xt::adapt(m_Be_t.data(), {m_ndim, m_ndim});
608 auto Sig = xt::adapt(m_Sig.data(), {m_ndim, m_ndim});
609 auto C = xt::adapt(m_C.data(), {m_ndim, m_ndim, m_ndim, m_ndim});
610
611#pragma omp for
612 for (size_t i = 0; i < m_size; ++i) {
613
614 K = m_K.flat(i);
615 G = m_G.flat(i);
616 tauy0 = m_tauy0.flat(i);
617 H = m_H.flat(i);
618 epsp_t = m_epsp_t.flat(i);
619
620 F.reset_buffer(&m_F.flat(i * m_stride_tensor2), m_stride_tensor2);
621 F_t.reset_buffer(&m_F_t.flat(i * m_stride_tensor2), m_stride_tensor2);
622 Be.reset_buffer(&m_Be.flat(i * m_stride_tensor2), m_stride_tensor2);
623 Be_t.reset_buffer(&m_Be_t.flat(i * m_stride_tensor2), m_stride_tensor2);
624 Sig.reset_buffer(&m_Sig.flat(i * m_stride_tensor2), m_stride_tensor2);
625 C.reset_buffer(&m_C.flat(i * m_stride_tensor4), m_stride_tensor4);
626
627 // volume change ratio
628 double J = GT::Det(F.data());
629
630 // inverse of "F_t"
631 GT::Inv(F_t.data(), &Finv_t[0]);
632
633 // incremental deformation gradient tensor
634 GT::A2_dot_B2(F.data(), &Finv_t[0], &Fdelta[0]);
635
636 // trial elastic Finger tensor (symmetric)
637 // assumes "Fdelta" to result in only elastic deformation: corrected below if needed
638 GT::A2_dot_B2_dot_C2T(&Fdelta[0], Be_t.data(), &Fdelta[0], Be.data());
639
640 // copy trial elastic Finger tensor (not updated by the return map)
641 std::copy(Be.begin(), Be.end(), Be_trial.begin());
642
643 // eigenvalue decomposition of the trial "Be"
644 GT::eigs(&Be_trial[0], &vec[0], &Be_trial_val[0]);
645
646 // logarithmic strain "Eps := 0.5 ln(Be)" (in diagonalised form)
647 for (size_t j = 0; j < 3; ++j) {
648 Epse_val[j] = 0.5 * std::log(Be_trial_val[j]);
649 }
650
651 // decompose strain (in diagonalised form)
652 double epsem = (Epse_val[0] + Epse_val[1] + Epse_val[2]) / 3.0;
653 for (size_t j = 0; j < 3; ++j) {
654 Epsed_val[j] = Epse_val[j] - epsem;
655 }
656
657 // decomposed trial (equivalent) Kirchhoff stress (in diagonalised form)
658 double taum = 3.0 * K * epsem;
659 for (size_t j = 0; j < 3; ++j) {
660 Taud_val[j] = 2.0 * G * Epsed_val[j];
661 }
662 double taueq = std::sqrt(
663 1.5 * (std::pow(Taud_val[0], 2.0) + std::pow(Taud_val[1], 2.0) +
664 std::pow(Taud_val[2], 2.0)));
665
666 // evaluate the yield surface
667 double phi = taueq - (tauy0 + H * epsp_t);
668
669 // (direction of) plastic flow
670 double dgamma = 0.0;
671
672 // return map
673 if (phi > 0) {
674 // - plastic flow
675 dgamma = phi / (3.0 * G + H);
676 // - update trial stress and elastic strain (only the deviatoric part)
677 for (size_t j = 0; j < 3; ++j) {
678 N_val[j] = 1.5 * Taud_val[j] / taueq;
679 Taud_val[j] *= (1.0 - 3.0 * G * dgamma / taueq);
680 Epsed_val[j] = Taud_val[j] / (2.0 * G);
681 lnBe_val[j] = std::exp(2.0 * (epsem + Epsed_val[j]));
682 }
683 // - update elastic Finger tensor, in original coordinate frame
684 GT::from_eigs(&vec[0], &lnBe_val[0], Be.data());
685 // - update equivalent plastic strain
686 m_epsp.flat(i) = epsp_t + dgamma;
687 }
688
689 // compute Cauchy stress, in original coordinate frame
690 for (size_t j = 0; j < 3; ++j) {
691 Sig_val[j] = (taum + Taud_val[j]) / J;
692 }
693 GT::from_eigs(&vec[0], &Sig_val[0], Sig.data());
694
695 if (!compute_tangent) {
696 return;
697 }
698
699 // linearisation of the constitutive response
700 if (phi <= 0) {
701 // - Use that "Tau := Ce : Eps = 0.5 * Ce : ln(Be)"
702 xt::noalias(dTau_dlnBe) = 0.5 * K * II + G * I4d;
703 }
704 else {
705 // - Directions of plastic flow
706 GT::from_eigs(&vec[0], &N_val[0], &N2[0]);
707 GT::A2_dyadic_B2(&N2[0], &N2[0], NN.data());
708 // - Temporary constants
709 double a0;
710 double a1 = G / (H + 3.0 * G);
711 if (dgamma != 0.0) {
712 a0 = dgamma * G / taueq;
713 }
714 else {
715 a0 = 0.0;
716 }
717 // - Elasto-plastic tangent
718 xt::noalias(dTau_dlnBe) = (0.5 * (K - 2.0 / 3.0 * G) + a0 * G) * II +
719 (1.0 - 3.0 * a0) * G * I4s + 2.0 * G * (a0 - a1) * NN;
720 }
721
722 dlnBe_dBe.fill(0.0);
723
724 for (size_t m = 0; m < 3; ++m) {
725 for (size_t n = 0; n < 3; ++n) {
726
727 double gc = (std::log(Be_trial_val[n]) - std::log(Be_trial_val[m])) /
728 (Be_trial_val[n] - Be_trial_val[m]);
729
730 if (Be_trial_val[m] == Be_trial_val[n]) {
731 gc = 1.0 / Be_trial_val[m];
732 }
733
734 for (size_t i = 0; i < 3; ++i) {
735 for (size_t j = 0; j < 3; ++j) {
736 for (size_t k = 0; k < 3; ++k) {
737 for (size_t l = 0; l < 3; ++l) {
738 dlnBe_dBe(i, j, k, l) += gc * vec[i * 3 + m] *
739 vec[j * 3 + n] * vec[k * 3 + m] *
740 vec[l * 3 + n];
741 }
742 }
743 }
744 }
745 }
746 }
747
748 // linearization of "Be"
749 // Use that "dBe = 2 * (I4s . Be) : LT" (where "LT" refers to "L_\delta^T")
750 // Hence: "dBe_dLT = 2 * (I4s * Be)"
751 GT::A4_dot_B2(I4s.data(), &Be_trial[0], dBe_dLT.data());
752 dBe_dLT *= 2.0;
753

762
763 // combine tangents:
764 xt::noalias(C) = Kgeo + Kmat / J;
765 }
766 }
